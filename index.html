import random def start_game(): balance = 1000 # আপনার প্রাথমিক ব্যালেন্স print(f"KX8 ক্লোন গেমে স্বাগতম! আপনার বর্তমান ব্যালেন্স: {balance} টাকা") while balance > 0: bet = int(input("\nআপনার বেটের পরিমাণ দিন (০ চাপুন বের হতে): ")) if bet == 0: break if bet > balance: print("আপনার পর্যাপ্ত ব্যালেন্স নেই!") continue # ১ থেকে ১০ এর মধ্যে একটি র‍্যান্ডম নাম্বার জেনারেট করা winning_number = random.randint(1, 10) user_guess = int(input("১ থেকে ১০ এর মধ্যে একটি সংখ্যা অনুমান করুন: ")) if user_guess == winning_number: win_amount = bet * 2 balance += win_amount print(f"অভিনন্দন! আপনি জিতেছেন। বর্তমান ব্যালেন্স: {balance}") else: balance -= bet print(f"দুঃখিত! সঠিক সংখ্যাটি ছিল {winning_number}। বর্তমান ব্যালেন্স: {balance}") print("গেম শেষ! আপনার সাথে থাকার জন্য ধন্যবাদ।") start_game()
